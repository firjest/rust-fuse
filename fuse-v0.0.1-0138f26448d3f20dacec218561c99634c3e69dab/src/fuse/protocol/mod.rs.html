macro_rules! op_mod {
	($name:ident, $path:expr) => {
		#[path = $path]
		pub mod $name;
		pub use self::$name::*;
	};
	($name:ident, $path:expr, $feature:expr) => {
		#[cfg(any(doc, feature = $feature))]
		#[cfg_attr(doc, doc(cfg(feature = $feature)))]
		#[path = $path]
		pub mod $name;
		#[cfg(any(doc, feature = $feature))]
		pub use self::$name::*;
	};
}
